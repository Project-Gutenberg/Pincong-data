//依赖的crate:rand rayon plotters eyre
use plotters::prelude::\*;
use rayon::prelude::\*;

fn task() -> (u64, u64) {
    let (mut boys, mut girls) = (0, 0);
    for \_ in 0..10 {
        //每家最多生10个，生到男的就停
        let is\_boy: bool = rand::random();
        if is\_boy {
            boys += 1;
            break;
        } else {
            girls += 1;
        }
    }
    (boys, girls)
}

const TASKS\_AMOUNT: u64 = 10000\_0000;

fn main() {
    let result: (u64, u64) = (0..TASKS\_AMOUNT)
        .into\_par\_iter()
        .map(|\_| task())
        .reduce\_with(|a, b| (a.0 + b.0, a.1 + b.1))
        .unwrap();

    dbg!(result);

    draw\_pie(result).unwrap();
}

//以下部分为绘制图表，不用看
const OUTPUT\_FILE: &str = "./output.png";
//这里可能要换成自己的中文字体
const FONT: &str = "LXGW Wenkai";
fn draw\_pie((boys, girls): (u64, u64)) -> eyre::Result<()> {
    let backend = BitMapBackend::new(OUTPUT\_FILE, (1000, 1000));
    let root\_area = backend.into\_drawing\_area();
    root\_area.fill(&WHITE)?;

    let title\_style = TextStyle::from((FONT, 50));
    root\_area.titled(
        &format!("结果:男女比例 样本数量:{}", TASKS\_AMOUNT),
        title\_style,
    )?;

    let (width, height) = root\_area.dim\_in\_pixel();
    let center = (width as i32 / 2, height as i32 / 2);
    let radius = 300.;

    let boys = boys as f64;
    let girls = girls as f64;
    let sum = boys + girls;
    let sizes = vec!\[boys / sum, girls / sum\];

    let colors = vec!\[BLUE, RED\];
    let labels = vec!\[format!("男:{}", boys), format!("女:{}", girls)\];

    let mut pie = Pie::new(¢er, &radius, &sizes, &colors, &labels);
    pie.start\_angle(66.);
    pie.label\_style((FONT, 30).into\_font());
    pie.percentages((FONT, 50).into\_font());
    root\_area.draw(&pie)?;

    Ok(())
}

### 知乎用户 知乎用户 8gYjll​ 发表
    
刷到这个问题很多次了，来答一下。

首先假设一个社群中每家每户在生小孩的过程中没有进行筛选性别的操作（譬如其他回答说的那些 b 超偷偷查性别、弃养之类的），每个新生儿的性别都是完全随机的；且我们假设每家人都会一直生小孩直到生出一个男孩为止。那么这个群体新生儿的男女比的[期望值](https://zhida.zhihu.com/search?content_id=650171913&content_type=Answer&match_order=1&q=%E6%9C%9F%E6%9C%9B%E5%80%BC&zhida_source=entity)是多少呢？

这其实是一个[概率统计](https://zhida.zhihu.com/search?content_id=650171913&content_type=Answer&match_order=1&q=%E6%A6%82%E7%8E%87%E7%BB%9F%E8%AE%A1&zhida_source=entity)里一个非常经典的问题，小绿书 [\[1\]]() 里就有这个问题的解答（不过是性转版本的）。简而言之就是，在自然状态下，你生小孩的意愿和偏好完全不影响你每次生小孩的性别概率分布，同时每次生小孩都是互相独立的事件，所以不管你生几次，以什么策略来生，男女比都是 1:1。

![](https://images.weserv.nl/?url=https%3A//picx.zhimg.com/v2-9e6b3d36f1ecf3583bd1f0b1d2c85f83_r.jpg%3Fsource%3D1def8aca)

这样说可能会让人觉得有点脑筋急转弯，我们来整点公式算一下：

假设整个社群中有 N 户家庭，那么我们可以根据生育小孩的数目做一个分类讨论：

1.  第一胎就生出男孩不再生的家庭是 N/2 户，一共贡献 N/2 个小孩，其中有 N/2 个男孩；
2.  第二胎生出男孩的家庭是 N/4 户，一共贡献 N/4\\times2=N/2 个小孩，其中有 N/4 个男孩；
3.  …. 以此类推，第 k 胎生出男孩的家庭是 N/2^k 户，一共贡献 k\\times N /2^k 个小孩，其中有 N/2^k 个男孩。

那么男孩的数目就是 \\sum\_{k=1}^\\infty \\frac{N}{2^k} = N 个；而所有小孩的数目是 \\sum\_{k=1}^\\infty N\\frac{k}{2^k} ，这个[级数](https://zhida.zhihu.com/search?content_id=650171913&content_type=Answer&match_order=1&q=%E7%BA%A7%E6%95%B0&zhida_source=entity)用一些小 trick 可以算出来是 2N 个；因此我们可以看到，即使这个 toy model 中充满了 “多姐一弟” 的家庭，男孩还是占到了新生儿的一半，[性别比](https://zhida.zhihu.com/search?content_id=650171913&content_type=Answer&match_order=1&q=%E6%80%A7%E5%88%AB%E6%AF%94&zhida_source=entity)依旧是 **1:1**。

最后，考虑到现实世界中渴望生男孩的家庭，还会采用各种手段筛选性别，由此增加生出男孩的概率，所以男女比必然大于等于 1。
